        private void button1_Click(object sender, EventArgs e)
        {
            int uzun = Convert.ToInt32(textBox1.Text);
            int kisa = Convert.ToInt32(textBox2.Text);
            Dikdortgen dikdortgen = new Dikdortgen();
            label3.Text = (dikdortgen.AlanHesapla(uzun, kisa)).ToString();
        }
        public class Dikdortgen
        {
            public int AlanHesapla(int uzun,int kisa)
            {
                return uzun * kisa;
            }
            public int CevreHesapla(int uzun, int kisa)
            {
                return 2*(uzun + kisa);
            }
        }

        private void button2_Click(object sender, EventArgs e)
        {
            int uzun = Convert.ToInt32(textBox1.Text);
            int kisa = Convert.ToInt32(textBox2.Text);
            Dikdortgen dikdortgen = new Dikdortgen();
            label3.Text = (dikdortgen.CevreHesapla(uzun, kisa)).ToString();
        }
